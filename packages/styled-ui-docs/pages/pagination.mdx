# Pagination

Pagination

## Import

```js
import Pagination from '@trendmicro/react-styled-ui/Pagination';
// or
import { Pagination } from '@trendmicro/react-styled-ui';
```

## Usage

```jsx
<Pagination count={10} />
```

### Disable

```jsx
<Pagination count={10} disabled />
```

### Default page

```jsx
<Pagination count={10} defaultPage={5} />
```

### On change

```jsx
<Pagination
  count={10}
  onChange={(event, value) => alert(`${event} ${value}`)}
/>
```

### Boundary count

```jsx
<Pagination count={20} boundaryCount={2} />
```

### Sibling count

```jsx
<Pagination
  count={20}
  siblingCount={2}
  defaultPage={10}
/>
```

### Full mode

```jsx
function Minor() {
  const [page, setPage] = React.useState(1);
  const totalPages = 20;
  const prevPageDisabled = (page <= 1);
  const nextPageDisabled = (page >= totalPages);

  return (
    <Flex>
      <Flex mr="3x" alignItems="center">
        <Text>
          Total: 400
        </Text>
        <Divider vertical height="6x" ml="3x" />
        <Menu>
          <MenuButton variant="ghost">10 per page</MenuButton>
          <MenuList>
            <MenuItem>10</MenuItem>
            <MenuItem>50</MenuItem>
            <MenuItem>100</MenuItem>
          </MenuList>
        </Menu>
        <Divider vertical height="6x" />
      </Flex>
      <Pagination count={totalPages} onChange={(_, value) => setPage(value)} />
    </Flex>
  );
}

```

### Minor mode

```jsx
function Minor() {
  const [page, setPage] = React.useState(1);
  const totalPages = 10;
  const prevPageDisabled = (page <= 1);
  const nextPageDisabled = (page >= totalPages);

  return (
    <Flex>
      <Flex mr="3x" alignItems="center">
        <Text>
          Total: 400
        </Text>
        <Divider vertical height="6x" mx="3x" />
        <Text>
          100 per page
        </Text>
        <Divider vertical height="6x" mx="3x" />
        <Text>{page}</Text>
        <Space width="2x" />
        <Text>/</Text>
        <Space width="2x" />
        <Text>{totalPages}</Text>
      </Flex>
      <ButtonGroup variant="secondary">
        <Button
          disabled={prevPageDisabled}
          onClick={(event) => {
            const prevPage = page > 1 ? page - 1 : page;
            if (prevPage !== page) {
              setPage(prevPage);
            }
          }}
        >
          <TMIcon name="angle-left" />
        </Button>
        <Button
          disabled={nextPageDisabled}
          onClick={(event) => {
            const nextPage = page < totalPages ? page + 1 : page;
            if (nextPage !== page) {
              setPage(nextPage);
            }
          }}
        >
          <TMIcon name="angle-right" />
        </Button>
      </ButtonGroup>
    </Flex>
  );
}
```
